22struct TableEntry
23{
24 uint64_t id{0};
25 double val{0.0};
26
27 TableEntry() = default;
28 TableEntry(uint64_t _id, double _val)
29 : id(_id), val(_val) { }
30};

32// top k bucket entry
33struct PQComparison
34{
35 bool operator() (const TableEntry &lhs, const TableEntry &rhs) const{
36 return lhs.val > rhs.val;
37 }
38};
